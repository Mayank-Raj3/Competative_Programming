void solve()
{
    int n, k;
    cin >> n >> k;
    vi v(n);
    REP(i, 0, n)
    cin >> v[i];
    auto check = [&](int mid)
    {
        return v[mid] >= k;
    };
    int low = 0, high = n - 1;
    int ans = high;
    while (low <= high)
    {
        int mid = (low + high) / 2;
        if (check(mid))
        {
            ans = mid;
            high = mid - 1;
        }
        else
            low = mid + 1;
    }
    cout << ans << "\n";
}

vi primes;
void precomp() {
    for (int i = 2; i <= 10000000; i++) {
        sieve[i] = 1;
    }
    for (int i = 2; i * i <= 10000000; i++) {
        if (sieve[i] == 0) continue;
        for (int j = i * i; j <= 10000000; j += i) {
            sieve[j] = 0; // making multiples as 0 since the will definetly will be not be prime
        }
    }
    for (int i = 2; i <= 10000000; i++) {
        if (sieve[i] == 1) {
            primes.pb(i);
        }
    }
}

int computeXOR(int n)
{
   
  // If n is a multiple of 4
  if (n % 4 == 0)
    return n;
 
  // If n%4 gives remainder 1
  if (n % 4 == 1)
    return 1;
 
  // If n%4 gives remainder 2
  if (n % 4 == 2)
    return n + 1;
 
  // If n%4 gives remainder 3
  return 0;
}
